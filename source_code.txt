#include <WiFi.h>
#include <HTTPClient.h>
#include <WiFiClientSecure.h>

// 1. Konfigurasi Jaringan WiFi Wokwi (Virtual WiFi)
const char* ssid = "Wokwi-GUEST";
const char* password = ""; 

// 2. Konfigurasi Blynk HTTPS REST API
const char* blynkToken = "yb-ZTUP9H748w2ac0ajPMkCCj1WmJRh9"; 
const char* eventCode = "maling_terdeteksi";

// 3. Definisi Pin Komponen
const int pinPIR = 12;
const int pinLED = 14;
const int pinBuzzer = 27;

void setup() {
  Serial.begin(115200);
  
  // Inisialisasi Mode Pin
  pinMode(pinPIR, INPUT);
  pinMode(pinLED, OUTPUT);
  pinMode(pinBuzzer, OUTPUT);

  // Memulai Koneksi WiFi
  WiFi.begin(ssid, password);
  Serial.print("Menghubungkan ke WiFi Virtual Wokwi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\n[Sukses] WiFi Terhubung!");
}

void loop() {
  // Membaca status sensor PIR (HIGH jika ada gerakan, LOW jika tidak)
  int statusPIR = digitalRead(pinPIR);

  if (statusPIR == HIGH) {
    Serial.println("\n[PERINGATAN] Gerakan Terdeteksi! Mengaktifkan Alarm Lokal...");
    
    // Aksi Aktuator Lokal (Lampu berkedip / menyala & Buzzer berbunyi)
    digitalWrite(pinLED, HIGH);
    tone(pinBuzzer, 1000); // Mengeluarkan suara frekuensi 1000Hz

    // Aksi Jaringan Cloud: Mengirimkan Notifikasi via HTTPS Client
    kirimNotifikasiBlynk();

    // Loop penahan: Menunggu sampai gerakan hilang agar tidak spam notifikasi
    while(digitalRead(pinPIR) == HIGH) {
      delay(500); 
    }
    
    // Mematikan alarm lokal setelah area kembali aman
    digitalWrite(pinLED, LOW);
    noTone(pinBuzzer);
    Serial.println("[Aman] Gerakan hilang. Menormalkan Sistem.");
  }
  delay(200); // Delay kecil untuk stabilitas pembacaan sensor
}

// Fungsi Khusus untuk melakukan Protokol HTTPS Web Client ke Blynk Cloud
void kirimNotifikasiBlynk() {
  WiFiClient client; // Menggunakan WiFiClient biasa (Ringan & bebas SSL)
  HTTPClient http;
  
  // KUNCI UTAMA: Menggunakan http:// dengan subdomain sgp1 dan PORT 8080 (:8080)
  String url = "http://sgp1.blynk.cloud:8080/external/api/logEvent?token=" + String(blynkToken) + "&code=" + String(eventCode);
  
  Serial.println("[HTTP] Mencoba mengirim request via Port 8080 ke Blynk Server...");
  
  if (http.begin(client, url)) {
    int httpCode = http.GET(); // Eksekusi request secara instan
    
    if (httpCode > 0) {
      Serial.printf("[HTTP] Server Merespon! Kode HTTP: %d\n", httpCode);
      if (httpCode == 200) {
        Serial.println("[Sukses] Notifikasi darurat berhasil diteruskan ke aplikasi HP!");
      }
    } else {
      Serial.printf("[Error] Gagal, keterangan: %s\n", http.errorToString(httpCode).c_str());
    }
    http.end(); 
  } else {
    Serial.println("[Error] Tidak dapat terhubung ke server.");
  }
}
